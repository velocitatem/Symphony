40std::shared_ptr<Node> BreadthFirstSearch::search() {
41 std::queue<std::shared_ptr<Node>> frontier;
42 frontier.push(std::make_shared<Node>(nullptr, problem->initial_state(), nullptr, 0, 0));
43 while (!frontier.empty()) {
44 auto node = std::move(frontier.front());
45 frontier.pop();
46 if (problem->goal_test(node->state)) {
47 return node;
48 }
49 for (auto &action : problem->actions(node->state)) {
50 auto child = std::make_shared<Node>(node.get(), action.effect, &action, node->path_cost + action.cost, problem->heuristic(action.effect));
51 frontier.push(std::move(child));
52 }
53 }
54 return nullptr;
